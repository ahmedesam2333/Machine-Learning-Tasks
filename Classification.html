4. Handilng the outliers¶

In [11]:
Q1 = df['Purchase_Frequency'].quantile(0.25)
Q3 = df['Purchase_Frequency'].quantile(0.75)

IQR = Q3-Q1 

LowerBound = Q1 - 1.5 * IQR
UpperBound = Q3 + 1.5 * IQR

df['Purchase_Frequency'] = np.where(df['Purchase_Frequency'] > UpperBound, UpperBound,  np.where(df['Purchase_Frequency'] < LowerBound, LowerBound, df['Purchase_Frequency']))
