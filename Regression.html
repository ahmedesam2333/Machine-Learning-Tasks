4- Handilng the outliers

In[10]:
columns_to_process = numerical_features
for col in columns_to_process:
    Q1 = reg_data[col].quantile(0.25)
    Q3 = reg_data[col].quantile(0.75)
    IQR = Q3 - Q1

    LowerBound = Q1 - 1.5 * IQR
    UpperBound = Q3 + 1.5 * IQR

    reg_data[col] = np.where(
        reg_data[col] > UpperBound, UpperBound,
        np.where(reg_data[col] < LowerBound, LowerBound, reg_data[col])
    )
  # Visualize the column after handling
    plt.figure(figsize=(8, 4))
    sns.boxplot(x=reg_data[col], color='blue', showfliers=True)
    sns.stripplot(x=reg_data[col], color='red', size=6, jitter=True)
    plt.title(f"{col} After Handling Outliers")
    plt.xlabel(col)
    plt.show()
